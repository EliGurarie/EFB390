\frame[t]{\frametitle{Метод Квадратов: Распределение оценки учета $\widehat{N}$}
\footnotesize
{\bf Показатели оценки:} \\
	$\bullet$ Среднее (\emph{mean}) или Ожидаемая величина (\emph{expected value/expectation})
	\begin{equation*} 
			\gr{\text{E}(N) = \mu = {1\over n} \sum\limits_{i=1}^n N_i}
	\end{equation*}
	
	$\bullet$ Медиана (\emph{median}): 
	\begin{equation*} 
			\gr{\widetilde\mu = Q_{0.5} (N_i)}
	\end{equation*}

{\bf Показатели ошибки:} \\
$\bullet$ Дисперсия (\emph{variance}): 
		\begin{equation*} 
			\gr{\text{Var}[N] = \sigma^2 = {1\over n-1} \sum\limits_{i=1}^n (N_i - \mu)^2} 
		\end{equation*}
$\bullet$ Стандартное отклонение (\emph{standard deviation} - SD): 
		\begin{equation*} 
			\gr{\sigma = \sqrt{\sigma^2}} 
		\end{equation*}
$\bullet$ Коэффицент вариации (\emph{coefficient of variation} - CV): 
	\begin{equation*} 
			\gr{\sigma / \mu}
	\end{equation*}
$\bullet$ Доверительный интервал (\emph{confidence interval} - CI): 
		\begin{equation*} 
				\gr{Q_{.025} (N_i), Q_{.975} (N_i) \approx \mu \pm 2\sigma_N}
		\end{equation*}
}

\begin{frame}[c]\frametitle{Уникальное свойство распределение Пуассона.}
			\includegraphics[width=\textwidth]{PoissonPlot.pdf}\\
	Если \gr{$X \sim \text{Poisson}(\lambda)$} то
			\rd{\begin{eqnarray*} 
					\text{E}[X] &=& \lambda \\ 
					\text{Var}[X] &=& \lambda
			\end{eqnarray*}}
		то есть: {\bf Дисперсия = Ожидание = Интенсивность!} \pause \\
		\normalsize Используя это свойство, легко можно проверить соответсвуют ли данные распределению Пуассона.
\end{frame}
